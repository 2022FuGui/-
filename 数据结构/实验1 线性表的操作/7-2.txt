#include<stdio.h>
int main()
{
    int a[10000];
    int x,y,i,t,n;
    int flag=1;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    scanf("%d%d",&x,&y);
    for (i=0;i<n;i++)
    {
        if(flag)
        {
            if(a[i]<x||a[i]>y)
            {
                printf("%d",a[i]);
                flag=0;
            }
        }
        else
        {
            if(a[i]<x||a[i]>y)
                printf(" %d",a[i]);
        }
    }
    return 0;
}
