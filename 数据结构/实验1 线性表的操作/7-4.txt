#include <iostream>
#include<string>
using namespace std;
 
int main()
{
    string str1,str2;
    getline(cin,str1);
    if(str1.size()==0)
    {
        cout<<"NULL";
        return 0;
    }
    for(int i = 0;i<str1.size();i++)
    {
       if(str1[i]>='A' && str1[i]<='E')
       {
           str1[i] = str1[i] + 21;
       }
       else if(str1[i]>='F' && str1[i] <= 'Z')
       {
           str1[i] = str1[i] - 5;
       }
    }
    cout << str1;
    return 0;
}
