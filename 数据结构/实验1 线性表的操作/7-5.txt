#include <iostream>
using namespace std;
int main(){
    int arr[100];
    int len,e,count=-1;
    cin >> len;
    for(int i=0;i<len;i++){
        cin >> arr[i];
    }
    cin >> e;
    for(int i=0;i<len;i++){
        if(arr[i]>=e){
            count = i;
            break;
        }
    }
    if(count==-1){
        arr[len] = e;
    } else{
        for(int k=len;k>count;k--){
            arr[k] = arr[k-1];
        }
        arr[count] = e;
    }
    len++;
    for(int i=0;i<len;i++){
        cout << arr[i];
        cout << ",";
    }

    return 0;
}
